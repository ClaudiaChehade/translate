The impact of disturbing surrounding camera on the stitching of BEV

Instructions for use

  • Support the external parameters, Pitch, ROLL, and Height, which disturb the external references

  • python3 disturb_nrcs_extrinsic.py -y yaml_folder -c camera_name[front/rear/left/right] -d delta_yaw/delta_pitch/delta_roll/delta_height[unit:degree/m]

  • When the T_V_C in the YAML file is rotated in order of Euler Corner XYZ, the pitch angle under the coordinate system of the vehicle corresponds to ROLL, so the disturbance of the Pitch needs to be applied on the delta_rolll

  • refer to View Camera BEV Stitching -Data Analysis Guide -Compatible with Shadow_mode, Parking, Line calibration and after -sales calibration data processing , Use the disturbed yaml file to generate BEV

import os
import cv2
import shutil
import argparse
import numpy as np
 
 
class DisturbExtrinsic():
def __init__(self, yaml_file, yaml_disturb, dof):
self.yaml_file = yaml_file
self.yaml_disturb = yaml_disturb
self.delta_yaw = dof[0]
self.delta_pitch = dof[1]
self.delta_roll = dof[2]
self.delta_height = dof[3]
 
def TransformToEulerAngle(self, transform):
scale = 180 / np.pi
roll = scale * np.arctan2(transform[2, 1], transform[2, 2])
pitch = scale * np.arctan2(-transform[2, 0], np.sqrt(transform[0, 0] * transform[0, 0] + transform[1, 0] * transform[1, 0]))
yaw = scale * np.arctan2(transform[1, 0], transform[0, 0])
return [roll, pitch, yaw]
 
def EulerAngle2Rotation(self, euler):
theta = [deg * np.pi / 180.0 for deg in euler]
R_x = np.array([[1, 0, 0],
[0, np.cos(theta[0]), -np.sin(theta[0])],
[0, np.sin(theta[0]), np.cos(theta[0])]])
R_y = np.array([[ np.cos(theta[1]), 0, np.sin(theta[1])],
[ 0, 1, 0 ],
[-np.sin(theta[1]), 0, np.cos(theta[1])]])
R_z = np.array([[np.cos(theta[2]), -np.sin(theta[2]), 0],
[np.sin(theta[2]), np.cos(theta[2]), 0],
[ 0, 0, 1]])
R = np.dot(R_z, np.dot(R_y, R_x))
return R
def change_transform(self):
fs_read = cv2.FileStorage(self.yaml_file, cv2.FileStorage_READ)
self.transform = fs_read.getNode('T_v_c').mat()
fs_read.release()
print("before disturb: \n", self.transform)
euler = self.TransformToEulerAngle(self.transform)
euler[0] += self.delta_roll
euler[1] += self.delta_pitch
euler[2] += self.delta_yaw
R = self.EulerAngle2Rotation(euler)
self.transform[:3, :3] = R
self.transform[2, 3] += self.delta_height
print("after disturb: \n", self.transform)
# -- keep intrinsic and update extrinsic
with open(self.yaml_file, 'r') as f_in:
with open(self.yaml_disturb, 'w') as f_out:
lines = f_in.readlines()
f_out.writelines(lines[:40])
f_out.write("T_v_c: !!opencv-matrix\n")
f_out.write(" rows: 4\n")
f_out.write(" cols: 4\n")
f_out.write(" dt: d\n")
f_out.write(" data: [ ")
for i in range(4):
for j in range(4):
f_out.write(f"{self.transform[i, j]:.5f}")
if i != 3 or j != 3:
f_out.write(", ")
f_out.write(" ]\n")
 
 
def copy_other_yaml_files(yaml_path, camera_name):
for name in ['front', 'rear', 'left', 'right']:
if name == camera_name:
continue
yaml_name = 'nrcs_' + name + '.yaml'
src_file = os.path.join(yaml_path, yaml_name)
dst_file = os.path.join(yaml_path, 'disturb', yaml_name)
shutil.copy(src_file, dst_file)
 
if __name__ == '__main__':
parser = argparse.ArgumentParser(description='disturb the extrinsic of nrcs')
parser.add_argument('-y', '--yaml_path', type=str, required=True)
parser.add_argument('-c', '--camera_name', type=str, required=True)
parser.add_argument('-d', '--disturb', type=str, required=True, help='yaw/pitch/roll/height')
 
args = parser.parse_args()
yaml_path = args.yaml_path
camera_name = args.camera_name
disturb = args.disturb.split('/')
if len(disturb) != 4:
raise ValueError('[ERROR] the disturb should be yaw/pitch/roll/height')
dof = [float(value) for value in disturb]
print(f'disturb the extrinsic of {camera_name}: yaw={dof[0]}degree pitch={dof[1]}degree roll={dof[1]}degree height={dof[3]}m')
yaml_name = 'nrcs_' + camera_name + '.yaml'
src_file = os.path.join(yaml_path, yaml_name)
dst_folder = os.path.join(yaml_path, 'disturb')
if os.path.exists(dst_folder):
shutil.rmtree(dst_folder)
os.makedirs(dst_folder)
dst_file = os.path.join(dst_folder, yaml_name)
disturb_extrinsic = DisturbExtrinsic(src_file, dst_file, dof)
disturb_extrinsic.change_transform()
copy_other_yaml_files(yaml_path, camera_name)
